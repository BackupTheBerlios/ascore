<?php

        function listAll($field,$addVoidValue=True,$more="") {

        if (empty($more))
            $all=$this->selectAll();
        else {
            setLimitRows(50);
            $all=$this->select($more);
            resetLimitRows();
        }
        
        if ($addVoidValue)
            $list[1]="--";
        foreach ($all as $k=>$o)
            if (method_exists($o,$field)) {
                debug("Llamada listAll tiene argumento a funcion","red");
                $list[$o->ID]=$o->$field();
            }
            else
                $list[$o->ID]=$o->$field;
        return $list;

    }
