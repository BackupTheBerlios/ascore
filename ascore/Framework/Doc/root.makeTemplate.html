<!--HEAD-->
<style  type="text/css">

</style>

';
        
        reset($this->properties_desc);

                 next($this->properties_desc);
                next($this->properties);
        }

<!--SET-->
<tr>
        ';

';

                $q.='

                $q.='

            else
                    $q.='
    <td class="<!-- dynamic_class -->"><!-- D:'.key($this->properties).' --></td>';

            next($this->properties_type);
            next($this->properties_desc);
            next($this->properties);
        }
    
        $q.='    
</tr>
<!--END-->
</table>
