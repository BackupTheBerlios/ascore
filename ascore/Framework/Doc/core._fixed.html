Wrapper para cademas, las amplia en size/2 por cada lado



<?php

    function _fixed($string,$len=17) {

    $l=strlen($string);
    if ($l<$len) {
        $iz=str_repeat("&nbsp;",($len-$l)/2);
        if ($l%2!=0)
            $de=str_repeat("&nbsp;",(($len-$l)/2)-1);
        else
            $de=$iz;
    }
    return "${iz}$string{$de}";
}


    ?>
